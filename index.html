参考命令

安装

wget http://res.szlt.net/master/lnmp.zip -P /tmp
cd /usr/local/src
unzip /tmp/lnmp.zip
chmod -R 777 sh && cd sh
./install.sh 2>&1 | tee install.log

卸载

cd /usr/local/src
./uninstall.sh 2>&1 | tee uninstall.log

重新安装

cd /usr/local/src
rm -rf sh
rm /tmp/lnmp.zip
wget http://res.szlt.net/master/lnmp.zip -P /tmp
unzip /tmp/lnmp.zip
chmod -R 777 sh && cd sh
./install.sh 2>&1 | tee install.log
